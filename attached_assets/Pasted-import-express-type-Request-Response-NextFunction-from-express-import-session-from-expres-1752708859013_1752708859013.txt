import express, { type Request, Response, NextFunction } from "express";

// Initialisation de l'application
const app = express();
app.use(express.json({ limit: "50mb" }));
app.use(express.urlencoded({ extended: false, limit: "50mb" }));

// Configuration de la session
app.use(
  session({
    secret: process.env.SESSION_SECRET || "spark-dev-secret-key-12345",
    resave: false,
    saveUninitialized: false,
    cookie: {
      secure: process.env.NODE_ENV === "production", // True en production avec HTTPS
      httpOnly: true,
      maxAge: 24 * 60 * 60 * 1000, // 24 hours
      sameSite: "lax", // SÃ©curitÃ© contre CSRF
    },
  })
);

// Middleware de logging amÃ©liorÃ©
app.use((req: Request, res: Response, next: NextFunction) => {
  const start = Date.now();
  const path = req.path;
  let capturedJsonResponse: Record<string, any> | undefined = undefined;

  const originalResJson = res.json;
  res.json = function (bodyJson, ...args) {
    capturedJsonResponse = bodyJson;
    return originalResJson.apply(res, [bodyJson, ...args]);
  };

  res.on("finish", () => {
    const duration = Date.now() - start;
    if (path.startsWith("/api")) {
      let logLine = `${req.method} ${path} ${res.statusCode} in ${duration}ms`;
      if (capturedJsonResponse) {
        logLine += ` :: ${JSON.stringify(capturedJsonResponse).slice(0, 50)}...`; // Limite Ã  50 chars
      }
      log(logLine.length > 80 ? logLine.slice(0, 79) + "â€¦" : logLine);
    }
  });

  next();
});

    log(`Erreur ${status}: ${message}`);
  });

  // Configuration WebSocket pour donnÃ©es en temps rÃ©el

  const io = new SocketIOServer(httpServer, {

      methods: ["GET", "POST"],
    },
  });

  io.on("connection", (socket) => {
    console.log("Utilisateur connectÃ© via WebSocket");
    socket.on("investmentUpdate", (data) => {
      io.emit("investmentUpdate", data); // Diffuse les mises Ã  jour
      log(`Mise Ã  jour investissement Ã©mise: ${JSON.stringify(data).slice(0, 50)}...`);

      console.log("Utilisateur dÃ©connectÃ© via WebSocket");

  // HTTPS en production (Ã  configurer avec certificats)
